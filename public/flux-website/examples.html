What Flux looks like

No semicolons. No imports. No boilerplate. You write what you mean and it runs.

Hello World
hello.flux 
fn main() {
    print("Hello, World!")
}
Structs
point.flux 
struct Point {
    x: Int
    y: Int
}

fn main() {
    let pt = Point { x: 10, y: 20 }
    print(pt.x + pt.y)  // 30
}
Functions
math.flux 
fn factorial(n: Int) -> Int {
    if (n <= 1) {
        return 1
    }
    return n * factorial(n - 1)
}

fn main() {
    print(factorial(10))
}
System Monitor
sysmon.flux 
fn main() {
    print("Host: " + sys_hostname())
    print("CPU:  " + sys_cpu_model())

    let used = sys_mem_used() / 1024
    let total = sys_mem_total() / 1024
    print("RAM:  " + str(used) + "/" + str(total) + " MB")
}
