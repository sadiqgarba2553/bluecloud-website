Flux Game Engine

1. Overview

Flux includes a built-in game engine with 2D and 3D capabilities, using an Entity Component System (ECS) architecture. It targets desktop, mobile, and web platforms.

2. Architecture

┌─────────────────────────────────────────────┐
│                 Game Loop                     │
│  Input → Update → Physics → Render → Audio   │
└─────────────────────────────────────────────┘
         │          │          │         │
    ┌────▼───┐ ┌────▼───┐ ┌───▼──┐ ┌───▼──┐
    │  ECS   │ │Physics │ │Render│ │Audio │
    │ World  │ │ Engine │ │Engine│ │Engine│
    └────────┘ └────────┘ └──────┘ └──────┘

3. Entity Component System

3.1 Components

use flux.game.*

// Components are plain data
struct Position {
    x: Float32
    y: Float32
}

struct Velocity {
    dx: Float32
    dy: Float32
}

struct Sprite {
    texture: Texture
    width: Float32
    height: Float32
}

struct Health {
    current: Int
    max: Int
}

struct Player {}  // Tag component (no data)
struct Enemy {}   // Tag component

3.2 Entities

fn spawn_player(world: &mut World) -> Entity {
    return world.spawn()
        .with(Position { x: 100.0, y: 100.0 })
        .with(Velocity { dx: 0.0, dy: 0.0 })
        .with(Sprite { texture: load_texture("player.png"), width: 32.0, height: 32.0 })
        .with(Health { current: 100, max: 100 })
        .with(Player {})
        .build()
}

fn spawn_enemy(world: &mut World, x: Float32, y: Float32) -> Entity {
    return world.spawn()
        .with(Position { x: x, y: y })
        .with(Velocity { dx: -1.0, dy: 0.0 })
        .with(Sprite { texture: load_texture("enemy.png"), width: 24.0, height: 24.0 })
        .with(Health { current: 50, max: 50 })
        .with(Enemy {})
        .build()
}

3.3 Systems

// Movement system
fn movement_system(query: Query<(&mut Position, &Velocity)>, dt: Float32) {
    for (pos, vel) in query {
        pos.x += vel.dx * dt
        pos.y += vel.dy * dt
    }
}

// Collision system
fn collision_system(
    players: Query<(&Position, &Sprite), With<Player>>,
    enemies: Query<(&Position, &Sprite, Entity), With<Enemy>>,
    commands: &mut Commands,
) {
    for (player_pos, player_sprite) in players {
        for (enemy_pos, enemy_sprite, enemy_entity) in enemies {
            if check_collision(player_pos, player_sprite, enemy_pos, enemy_sprite) {
                commands.despawn(enemy_entity)
                // Emit event, add score, etc.
            }
        }
    }
}

// Render system
fn render_system(query: Query<(&Position, &Sprite)>, renderer: &mut Renderer) {
    for (pos, sprite) in query {
        renderer.draw_sprite(sprite.texture, pos.x, pos.y, sprite.width, sprite.height)
    }
}

4. Game Loop

use flux.game.*

fn main() {
    let game = Game.new(GameConfig {
        title: "My Flux Game"
        width: 1280
        height: 720
        vsync: true
        target_fps: 60
    })

    game.add_system(SystemStage.Update, movement_system)
    game.add_system(SystemStage.Update, collision_system)
    game.add_system(SystemStage.Update, input_system)
    game.add_system(SystemStage.Render, render_system)
    game.add_system(SystemStage.Render, ui_render_system)

    game.add_startup_system(fn(world: &mut World) {
        spawn_player(world)
        for i in 0..10 {
            spawn_enemy(world, 400.0 + i as Float32 * 60.0, 200.0)
        }
    })

    game.run()
}

fn setup_3d_scene(scene: &mut Scene3D) {
    // Camera
    scene.add_camera(Camera3D {
        position: Vec3(0, 5, 10)
        look_at: Vec3.ZERO

    // Lights
    scene.add_light(DirectionalLight {

    })

    scene.add_light(PointLight {
        position: Vec3(0, 5, 0)
        color: Color.warm_white()
        intensity: 2.0
        range: 20.0

    // Meshes
    scene.spawn_mesh(Mesh.cube(size: 1.0))

        .material(Material.pbr(
            albedo: Color.hex("#8B4513")
            roughness: 0.7
            metallic: 0.0
        ))

    // Skybox
    scene.set_skybox(Skybox.from_hdri("sky.hdr"))
}

6. Physics

use flux.game.physics.*

// 2D Physics
struct RigidBody2D {
    body_type: BodyType  // Static, Dynamic, Kinematic
    mass: Float32
    friction: Float32
    restitution: Float32  // Bounciness
}

struct Collider2D {
    shape: ColliderShape  // Box, Circle, Polygon, Capsule
}

// Add physics to an entity
world.spawn()
    .with(Position { x: 100.0, y: 100.0 })
    .with(RigidBody2D { body_type: BodyType.Dynamic, mass: 1.0, friction: 0.3, restitution: 0.5 })
    .with(Collider2D { shape: ColliderShape.Circle(radius: 16.0) })
    .build()

// Collision events
fn on_collision(event: CollisionEvent) {
    match event {
        CollisionEvent.Begin(a, b) -> {
            print("Collision between {a} and {b}")
        }
        CollisionEvent.End(a, b) -> {
            print("Separation between {a} and {b}")
        }
    }
}

7. Input Handling

use flux.game.input.*

fn input_system(input: &Input, query: Query<(&mut Velocity), With<Player>>) {
    for vel in query {
        vel.dx = 0.0
        vel.dy = 0.0

        if input.key_held(Key.Left) or input.key_held(Key.A) {
            vel.dx = -200.0
        }
        if input.key_held(Key.Right) or input.key_held(Key.D) {
            vel.dx = 200.0
        }
        if input.key_pressed(Key.Space) {
            vel.dy = -400.0  // Jump
        }

        // Mouse
        if input.mouse_pressed(MouseButton.Left) {
            let (mx, my) = input.mouse_position()
            shoot_at(mx, my)
        }

        // Gamepad
        if let Some(gamepad) = input.gamepad(0) {
            vel.dx = gamepad.left_stick_x() * 200.0
            vel.dy = gamepad.left_stick_y() * 200.0
        }
    }
}

12. Networking (Multiplayer)

use flux.game.net.*

// Server
let server = GameServer.new(port: 7777, max_players: 16)

server.on_connect(fn(player: PlayerId) {
    print("Player {player} connected")
    server.send(player, SpawnMessage { ... })
})

server.on_message(fn(player: PlayerId, msg: GameMessage) {
    match msg {
        GameMessage.Move(direction) -> {
            update_player_position(player, direction)
            server.broadcast(PositionUpdate { player: player, ... })
        }
    }
})

// Client
let client = GameClient.connect("localhost:7777")

client.on_message(fn(msg: ServerMessage) {
    match msg {
        ServerMessage.Spawn(data) -> spawn_remote_player(data)
        ServerMessage.Position(data) -> update_remote_position(data)
    }
})

fn send_input(client: &GameClient, direction: Vec2) {
    client.send(GameMessage.Move(direction))
}
