Flux UI — Cross-Platform GUI Framework

1. Overview

Flux UI is a declarative, cross-platform GUI framework inspired by SwiftUI and Flutter. It compiles to native platform widgets, providing a native look and feel on every platform.

2. Basic Syntax

use flux.ui.*

fn main() {
    App.run(MyApp)
}

struct MyApp: View {
    fn body(self) -> View {
        Window(title: "Hello Flux UI") {
            VStack(spacing: 16) {
                Text("Welcome to Flux UI! ⚡")
                    .font(.title)
                    .color(.primary)

                Text("Build beautiful cross-platform apps")
                    .font(.body)
                    .color(.secondary)

                Button("Get Started") {
                    print("Button clicked!")
                }
                .style(.primary)
                .padding(12)
            }
            .padding(24)
            .alignment(.center)
        }
    }
}

4. State Management

4.1 State

struct Counter: View {
    @State mut count: Int = 0

    fn body(self) -> View {
        VStack {
            Text("Count: {self.count}")
                .font(.title)

            HStack(spacing: 16) {
                Button("-") { self.count -= 1 }
                Button("+") { self.count += 1 }
            }
        }
    }
}

4.2 Binding

struct ParentView: View {
    @State mut name: String = ""

    fn body(self) -> View {
        VStack {
            NameEditor(name: &self.name)
            Text("Hello, {self.name}!")
        }
    }
}

struct NameEditor: View {
    @Binding name: String

    fn body(self) -> View {
        TextField("Your name", text: &self.name)
    }
}

4.3 Observable

@Observable
class UserStore {
    mut users: List<User> = []
    mut is_loading: Bool = false

    async fn fetch_users(mut self) {
        self.is_loading = true
        self.users = await api.get_users()
        self.is_loading = false
    }
}

struct UserList: View {
    @ObservedObject store: UserStore

    fn body(self) -> View {
        VStack {
            if store.is_loading {
                ProgressSpinner()
            } else {
                List(store.users) { user in
                    Text(user.name)
                }
            }
        }
        .on_appear {
            await store.fetch_users()
        }
    }
}

4.4 Environment

// Define an environment value
struct Theme {
    primary_color: Color
    font_family: String
}

let dark_theme = Theme {
    primary_color: Color.hex("#BB86FC")
    font_family: "Inter"
}

// Provide it
App.run(MyApp)
    .environment(\.theme, dark_theme)

// Consume it
struct ThemedButton: View {
    @Environment(\.theme) theme: Theme
    label: String

    fn body(self) -> View {
        Button(label) { ... }
            .background(theme.primary_color)
            .font(.custom(theme.font_family, size: 16))
    }
}

5. Navigation

struct MyApp: View {
    fn body(self) -> View {
        NavigationStack {
            List(items) { item in
                NavigationLink(item.title, destination: DetailView(item: item))
            }
            .navigation_title("Items")
        }
    }
}

// Tab navigation
struct MainView: View {
    @State mut selected_tab: Int = 0

    fn body(self) -> View {
        TabView(selection: &selected_tab) {
            HomeView()
                .tab_item("Home", icon: .house)

                .tab_item("Search", icon: .magnifying_glass)
                .tag(1)

            ProfileView()
                .tab_item("Profile", icon: .person)
                .tag(2)
        }
    }
}

6. Styling and Theming

// Modifiers
Text("Hello")
    .font(.title)
    .color(.white)
    .background(.blue)
    .padding(16)
    .corner_radius(8)
    .shadow(radius: 4, color: .black.opacity(0.2))
    .border(.gray, width: 1)

// Custom styles
style PrimaryButton: ButtonStyle {
    fn make_body(config: ButtonConfig) -> View {
        Text(config.label)
            .font(.headline)
            .color(.white)
            .padding(horizontal: 24, vertical: 12)
            .background(.blue)
            .corner_radius(8)
            .scale(config.is_pressed ? 0.95 : 1.0)
            .animation(.spring)
    }
}

Button("Login")
    .button_style(PrimaryButton)

7. Animations

struct AnimatedView: View {
    @State mut is_expanded: Bool = false

    fn body(self) -> View {
        VStack {
            Rectangle()
                .fill(.blue)
                .frame(
                    width: is_expanded ? 300 : 100,
                    height: is_expanded ? 300 : 100,
                )
                .corner_radius(is_expanded ? 16 : 50)
                .animation(.ease_in_out(duration: 0.3))

            Button(is_expanded ? "Shrink" : "Expand") {
                self.is_expanded = not self.is_expanded
            }
        }
    }
}

8.1 Platform-Specific Code

#[cfg(target_os: "macos")]
fn show_native_dialog(title: String, message: String) {
    MacOS.alert(title: title, message: message)
}

#[cfg(target_os: "windows")]
fn show_native_dialog(title: String, message: String) {
    Windows.message_box(title: title, message: message)
}

#[cfg(target_os: "linux")]
fn show_native_dialog(title: String, message: String) {
    Gtk.dialog(title: title, message: message)
}

10. Hot Reload

During development, Flux UI supports hot reload:

flux ui dev

Changes to view code are reflected immediately without restarting the app. State is preserved across reloads.
