Flux Web Framework

1. Overview

Flux includes a built-in web framework for creating web applications and APIs. It provides both a declarative page syntax for full-stack web apps and a programmatic API for REST/GraphQL servers.

2. Declarative Pages

2.1 Page Syntax

use flux.web.{page, component}

page "/" {
    title "Home — My App"
    meta description: "Welcome to my Flux web application"

    nav {
        link "/" { text "Home" }
        link "/about" { text "About" }
        link "/contact" { text "Contact" }
    }

    main {
        h1 "Welcome to Flux Web"
        p "Build fast, modern web applications with Flux."

        button "Get Started" {
            on_click -> navigate("/docs")
        }
    }

    footer {
        text "© 2026 My App"
    }
}

2.2 Dynamic Pages

page "/users/{id}" {
    let user = await fetch_user(id)?

    title "{user.name} — Profile"

    main {
        h1 user.name
        p user.bio

        if user.posts.length > 0 {
            h2 "Posts"
            for post in user.posts {
                article {
                    h3 post.title
                    p post.excerpt
                    link "/posts/{post.id}" { text "Read more" }
                }
            }
        } else {
            p "No posts yet."
        }
    }
}

2.3 Components

component Card(title: String, children: Content) {
    div class: "card" {
        h3 class: "card-title" { text title }
        div class: "card-body" {
            children
        }
    }
}

component UserCard(user: User) {
    Card(title: user.name) {
        p user.email
        p "Joined: {user.created_at.format('MMM d, yyyy')}"
    }
}

page "/team" {
    title "Our Team"
    main {
        h1 "Team Members"
        for member in team_members {
            UserCard(user: member)
        }
    }
}

3. HTTP Server API

3.1 Basic Server

use flux.http.{Server, Request, Response, Status}

fn main() {
    let server = Server.new("0.0.0.0:8080")

    server.get("/", fn(req: Request) -> Response {
        return Response.html("<h1>Hello, Flux!</h1>")
    })

    server.get("/api/health", fn(req: Request) -> Response {
        return Response.json({"status": "ok", "version": "1.0"})
    })

    server.post("/api/users", fn(req: Request) -> Response {
        let body: CreateUserInput = req.json()?
        let user = create_user(body)?
        return Response.json(user).status(Status.Created)
    })

    print("Server running on http://localhost:8080")
    server.start()
}

3.2 Routing

use flux.web.{Router, Request, Response}

fn api_routes() -> Router {
    let router = Router.new()

    router.group("/api/v1", fn(r) {
        r.get("/users", list_users)
        r.get("/users/{id}", get_user)
        r.post("/users", create_user)
        r.put("/users/{id}", update_user)
        r.delete("/users/{id}", delete_user)

        r.group("/posts", fn(r) {
            r.get("/", list_posts)
            r.get("/{id}", get_post)
            r.post("/", create_post)
        })
    })

    return router
}

3.3 Middleware

use flux.web.{Middleware, Request, Response, Next}

fn logging_middleware(req: Request, next: Next) -> Response {
    let start = Time.now()
    let response = next(req)
    let duration = Time.now() - start
    print("{req.method} {req.path} → {response.status} ({duration}ms)")
    return response
}

fn auth_middleware(req: Request, next: Next) -> Response {
    let token = req.header("Authorization")
    match token {
        Some(t) -> {
            match verify_token(t) {
                Ok(user) -> {
                    req.set("user", user)
                    return next(req)
                }
                Err(_) -> return Response.status(401).json({"error": "Invalid token"})
            }
        }
        None -> return Response.status(401).json({"error": "Missing authorization"})
    }
}

// Apply middleware
server.use(logging_middleware)
server.group("/api", fn(r) {
    r.use(auth_middleware)
    r.get("/protected", protected_handler)
})

4. Rendering Modes

4.1 Server-Side Rendering (SSR)

Pages are rendered on the server and sent as HTML:

#[render(mode: "ssr")]
page "/products/{id}" {
    let product = await db.get_product(id)?
    // HTML is generated on the server
    main {
        h1 product.name
        p product.description
    }
}

4.2 Client-Side Rendering (CSR)

Pages are rendered in the browser via WebAssembly:

#[render(mode: "csr")]
page "/dashboard" {
    mut data = State.new(List<DataPoint>.new())

    on_mount {
        data.set(await fetch_dashboard_data())
    }

    main {
        h1 "Dashboard"
        Chart(data: data.get())
    }
}

4.3 Hydration

SSR with client-side hydration — the page is rendered on the server, then the client-side code takes over:

#[render(mode: "hydrate")]
page "/interactive" {
    mut count = State.new(0)

    main {
        h1 "Counter: {count.get()}"
        button "Increment" {
            on_click -> count.set(count.get() + 1)
        }
    }
}

5. Database ORM

use flux.web.db.{Model, migrate, query}

#[model(table: "users")]
struct User {
    #[primary_key(auto_increment)]
    id: Int

    #[unique]
    email: String

    name: String

    #[default("now()")]
    created_at: DateTime
}

// Queries
async fn get_users() -> Result<List<User>, DbError> {
    return await User.all()
}

async fn find_user(id: Int) -> Result<Option<User>, DbError> {
    return await User.find(id)
}

async fn create_user(name: String, email: String) -> Result<User, DbError> {
    return await User.create(User {
        name: name
        email: email
    })
}

async fn search_users(query: String) -> Result<List<User>, DbError> {
    return await User.where("name LIKE ?", "%{query}%")
        .order_by("created_at", "desc")
        .limit(20)
        .execute()
}

6. WebSocket Support

use flux.web.{WebSocket, WsMessage}

server.ws("/chat", fn(ws: WebSocket) {
    print("Client connected: {ws.id}")

    ws.on_message(fn(msg: WsMessage) {
        match msg {
            WsMessage.Text(text) -> {
                // Broadcast to all connected clients
                ws.broadcast(WsMessage.Text("User: {text}"))
            }
            WsMessage.Binary(data) -> {
                // Handle binary data
            }
            WsMessage.Close -> {
                print("Client disconnected: {ws.id}")
            }
        }
    })
})

7. Authentication

use flux.web.auth.{Auth, Session, OAuth}

// Session-based auth
let auth = Auth.session(SessionConfig {
    secret: env("SESSION_SECRET")
    cookie_name: "flux_session"
    max_age: Duration.days(7)
})

// OAuth
let google_auth = OAuth.google(OAuthConfig {
    client_id: env("GOOGLE_CLIENT_ID")
    client_secret: env("GOOGLE_CLIENT_SECRET")
    redirect_uri: "http://localhost:8080/auth/callback"
})

server.get("/auth/google", google_auth.login_handler())
server.get("/auth/callback", google_auth.callback_handler())

8. Compilation

The web framework compiles Flux pages to optimized output:

# Build for production
flux web build --output dist/

# Development server with hot reload
flux web dev --port 3000

Output structure:

dist/
├── index.html
├── about/index.html
├── assets/
│   ├── app.js          # Bundled JavaScript (from WASM bridge)
│   ├── app.wasm        # WebAssembly module (CSR/hydration)
│   ├── styles.css      # Extracted and minified CSS
│   └── images/
├── api/                # Server-side API routes
└── _flux/
    └── manifest.json   # Build manifest
