Flux Best Practices

Patterns and guidelines for writing production-quality Flux code.

1. Error Handling

Use Result<T, E> for Recoverable Errors

// Good — caller decides how to handle
fn parse_config(path: String) -> Result<Config, ConfigError> {
    let content = File.read(path)?
    let config = toml.parse(content)?
    return Ok(config)
}

// Bad — panics on failure
fn parse_config(path: String) -> Config {
    return toml.parse(File.read(path).unwrap()).unwrap()
}

Create Domain-Specific Error Types

enum AppError {
    Config(ConfigError),
    Database(DbError),
    Auth(AuthError),
    Validation(field: String, message: String),
}

impl Display for AppError {
    fn to_string(self) -> String {
        match self {
            Config(e)   -> "Configuration error: {e}"
            Database(e) -> "Database error: {e}"
            Auth(e)     -> "Authentication error: {e}"
            Validation(field, msg) -> "Validation error on '{field}': {msg}"
        }
    }
}

Add Context to Errors

fn setup_database() -> Result<Connection, AppError> {
    let config = load_config()
        .context("Failed to load database configuration")?

    let conn = Database.connect(config.db_url)
        .context("Failed to connect to database at {config.db_url}")?

    return Ok(conn)
}

2. API Design

Use the Builder Pattern for Complex Construction

struct HttpClientBuilder {
    timeout: Duration
    max_retries: Int
    headers: HashMap<String, String>
    base_url: Option<String>
}

impl HttpClientBuilder {
    fn new() -> HttpClientBuilder {
        return HttpClientBuilder {
            timeout: Duration.seconds(30)
            max_retries: 3
            headers: HashMap.new()
            base_url: None
        }
    }

    fn with_timeout(mut self, timeout: Duration) -> HttpClientBuilder {
        self.timeout = timeout
        return self
    }

    fn with_max_retries(mut self, retries: Int) -> HttpClientBuilder {
        self.max_retries = retries
        return self
    }

    fn with_base_url(mut self, url: String) -> HttpClientBuilder {
        self.base_url = Some(url)
        return self
    }

    fn build(self) -> HttpClient {
        return HttpClient.from_builder(self)
    }
}

// Usage
let client = HttpClientBuilder.new()
    .with_timeout(Duration.seconds(10))
    .with_max_retries(5)
    .with_base_url("https://api.example.com")
    .build()

Accept Broad Types, Return Specific Types

// Good — accepts anything Iterable, returns specific List
fn deduplicate<T: Eq + Hash>(items: impl Iterable<T>) -> List<T> {
    mut seen = HashSet<T>.new()
    mut result = List<T>.new()
    for item in items {
        if seen.insert(item.clone()) {
            result.append(item)
        }
    }
    return result
}

Make Invalid States Unrepresentable

// Bad — many invalid combinations possible
struct Connection {
    host: String
    port: Int
    is_connected: Bool
    socket: Option<Socket>
    error: Option<String>
}

// Good — each state is a distinct variant
enum Connection {
    Disconnected(host: String, port: Int),
    Connecting(host: String, port: Int),
    Connected(socket: Socket),
    Error(host: String, error: ConnectionError),
}

3. Concurrency

Use Structured Concurrency

// Good — all tasks complete before function returns
async fn process_batch(items: List<Item>) -> List<Result> {
    let group = TaskGroup<Result>.new()
    for item in items {
        group.spawn(async { return await process(item) })
    }
    return await group.collect()
}

// Bad — fire-and-forget tasks (no control over lifetime)
async fn process_batch_bad(items: List<Item>) {
    for item in items {
        spawn async { await process(item) }  // Who owns these tasks?
    }
}

Use Channels for Communication

// Good — communicate through channels
async fn pipeline(input: Receiver<Data>) -> Receiver<Output> {
    let (tx, rx) = Channel<Output>.bounded(100)
    spawn async {
        while let Some(data) = await input.receive() {
            let output = transform(data)
            await tx.send(output)
        }
        tx.close()
    }
    return rx
}

Limit Concurrency

async fn fetch_all(urls: List<String>) -> List<Response> {
    let sem = Semaphore.new(10)  // Max 10 concurrent requests
    let group = TaskGroup<Response>.new()

    for url in urls {
        let permit = await sem.acquire()
        group.spawn(async {
            defer permit.release()
            return await http.get(url)
        })
    }

    return await group.collect()
}

4. Performance

Prefer Stack Allocation

// Good — struct on the stack
struct Point { x: Float64, y: Float64 }
let p = Point { x: 1.0, y: 2.0 }

// Avoid — unnecessary heap allocation
let p = Box.new(Point { x: 1.0, y: 2.0 })

Use Iterators Instead of Intermediate Collections

// Good — lazy evaluation, no intermediate allocations
let total = items.iter()
    .filter((item) => item.is_active())
    .map((item) => item.value)
    .sum()

// Avoid — creates intermediate collections
let active = items.filter((item) => item.is_active())
let values = active.map((item) => item.value)
let total = values.sum()

Clone Deliberately

// Good — explicit about cloning
let config_copy = config.clone()
modify(config_copy)

// Good — pass by reference when not modifying
fn display(config: &Config) {
    print(config.name)
}

5. Testing

Test Behavior, Not Implementation

#[test]
fn test_user_registration() {
    mut service = UserService.new()

    let result = service.register("alice@test.com", "password123")
    assert(result.is_ok())

    let user = result.unwrap()
    assert(user.email == "alice@test.com")
    assert(user.is_active)
}

Use Table-Driven Tests

#[test]
fn test_parse_int() {
    let cases = [
        ("42", Ok(42)),
        ("-1", Ok(-1)),
        ("0", Ok(0)),
        ("abc", Err(ParseError.InvalidFormat)),
        ("", Err(ParseError.EmptyInput)),
        ("99999999999999999999", Err(ParseError.Overflow)),
    ]

    for (input, expected) in cases {
        let result = parse_int(input)
        assert(result == expected, "parse_int(\"{input}\") = {result}, expected {expected}")
    }
}

Test Error Cases

#[test]
fn test_divide_by_zero() {
    let result = divide(10, 0)
    assert(result.is_err())
    assert(result.unwrap_err() == MathError.DivisionByZero)
}

6. Security

Validate All Input

fn create_user(input: CreateUserInput) -> Result<User, ValidationError> {
    if input.email.is_empty() {
        return Err(ValidationError.required("email"))
    }
    if not is_valid_email(input.email) {
        return Err(ValidationError.invalid("email", "must be a valid email address"))
    }
    if input.password.length < 8 {
        return Err(ValidationError.invalid("password", "must be at least 8 characters"))
    }
    // ... create user
}

Use Type-Safe IDs

struct UserId { value: String }
struct PostId { value: String }

// Prevents accidentally passing a PostId where a UserId is expected
fn get_user(id: UserId) -> Result<User, DbError> { ... }
fn get_post(id: PostId) -> Result<Post, DbError> { ... }

Don't Log Sensitive Data

fn authenticate(email: String, password: String) -> Result<Token, AuthError> {
    log.info("Authentication attempt for {email}")
    // Never: log.info("Password: {password}")

    let user = find_user_by_email(email)?
    if not verify_password(password, user.password_hash) {
        log.warn("Failed authentication for {email}")
        return Err(AuthError.InvalidCredentials)
    }

    log.info("Successful authentication for {email}")
    return Ok(generate_token(user))
}

7. Project Organization

Keep Modules Focused

src/
├── main.flux           // Entry point — thin, delegates to modules
├── config.flux         // Configuration loading and validation
├── models/
│   ├── mod.flux        // Re-exports
│   ├── user.flux       // User struct and methods
│   └── post.flux       // Post struct and methods
├── services/
│   ├── mod.flux
│   ├── auth.flux       // Authentication logic
│   └── email.flux      // Email sending
├── handlers/
│   ├── mod.flux
│   ├── user_handler.flux
│   └── post_handler.flux
└── utils/
    ├── mod.flux
    ├── validation.flux
    └── crypto.flux

Minimize Public Surface

// In mod.flux — only export what external users need
pub use self.user.User
pub use self.user.UserRole

// Don't export internal helpers
// self.user.validate_email stays private
