Flux Tutorial — Build a Task Manager CLI

This tutorial teaches Flux by building a complete command-line task manager application. By the end, you'll understand variables, functions, structs, enums, traits, error handling, file I/O, collections, and testing.

Chapter 3: Structs and Methods

use flux.time.DateTime

enum Priority {
    Low,
    Medium,
    High,
    Critical,
}

struct Task {
    id: Int
    title: String
    description: String
    priority: Priority
    completed: Bool
    created_at: DateTime
}

impl Task {
    fn new(id: Int, title: String, priority: Priority) -> Task {
        return Task {
            id: id
            title: title
            description: ""
            priority: priority
            completed: false
            created_at: DateTime.now()
        }
    }

    fn complete(mut self) {
        self.completed = true
    }

    fn display(self) -> String {
        let status = if self.completed { "✅" } else { "⬜" }
        let prio = match self.priority {
            Priority.Low      -> "🟢"
            Priority.Medium   -> "🟡"
            Priority.High     -> "🟠"
            Priority.Critical -> "🔴"
        }
        return "{status} [{self.id}] {prio} {self.title}"
    }
}

fn main() {
    mut task = Task.new(1, "Learn Flux", Priority.High)
    print(task.display())   // ⬜ [1] 🟠 Learn Flux

    task.complete()
    print(task.display())   // ✅ [1] 🟠 Learn Flux
}

Chapter 4: Collections and Iteration

struct TaskManager {
    tasks: List<Task>
    next_id: Int
}

impl TaskManager {
    fn new() -> TaskManager {
        return TaskManager { tasks: List.new(), next_id: 1 }
    }

    fn add_task(mut self, title: String, priority: Priority) -> Int {
        let id = self.next_id
        let task = Task.new(id, title, priority)
        self.tasks.append(task)
        self.next_id += 1
        return id
    }

    fn complete_task(mut self, id: Int) -> Result<Void, String> {
        for mut task in &mut self.tasks {
            if task.id == id {
                task.complete()
                return Ok(())
            }
        }
        return Err("Task #{id} not found")
    }

    fn list_all(self) {
        if self.tasks.is_empty() {
            print("No tasks yet!")
            return
        }

        print("--- Tasks ---")
        for task in &self.tasks {
            print(task.display())
        }
        print("--- {self.tasks.length} task(s) ---")
    }

    fn list_pending(self) -> List<&Task> {
        return self.tasks.iter()
            .filter(fn(t) { not t.completed })
            .collect()
    }

    fn remove_task(mut self, id: Int) -> Result<Task, String> {
        let index = self.tasks.iter()
            .position(fn(t) { t.id == id })

        match index {
            Some(i) -> return Ok(self.tasks.remove(i))
            None    -> return Err("Task #{id} not found")
        }
    }
}

fn main() {
    mut manager = TaskManager.new()

    manager.add_task("Buy groceries", Priority.Medium)
    manager.add_task("Write report", Priority.High)
    manager.add_task("Fix bug #42", Priority.Critical)
    manager.add_task("Read book", Priority.Low)

    manager.list_all()

    // Complete a task
    match manager.complete_task(2) {
        Ok(())   -> print("\nTask #2 completed!")
        Err(msg) -> print("Error: {msg}")
    }

    // List pending tasks
    print("\n--- Pending ---")
    for task in manager.list_pending() {
        print(task.display())
    }
}

Chapter 5: Enums and Pattern Matching

enum Command {
    Add(title: String, priority: Priority),
    Complete(id: Int),
    Remove(id: Int),
    List,
    ListPending,
    Help,
    Quit,
}

fn parse_command(input: String) -> Result<Command, String> {
    let parts = input.trim().split(" ")
    let cmd = parts[0].to_lower()

    match cmd {
        "add" -> {
            if parts.length < 2 {
                return Err("Usage: add <title> [priority]")
            }
            let title = parts[1..].join(" ")
            return Ok(Command.Add(title: title, priority: Priority.Medium))
        }
        "done" -> {
            if parts.length < 2 {
                return Err("Usage: done <id>")
            }
            let id = parts[1].parse_int()?
            return Ok(Command.Complete(id: id))
        }
        "remove" | "rm" -> {
            if parts.length < 2 {
                return Err("Usage: remove <id>")
            }
            let id = parts[1].parse_int()?
            return Ok(Command.Remove(id: id))
        }
        "list" | "ls"   -> return Ok(Command.List)
        "pending"       -> return Ok(Command.ListPending)
        "help" | "?"    -> return Ok(Command.Help)
        "quit" | "exit" -> return Ok(Command.Quit)
        _               -> return Err("Unknown command: {cmd}. Type 'help' for commands.")
    }
}

fn execute_command(manager: &mut TaskManager, cmd: Command) -> Bool {
    match cmd {
        Command.Add(title, priority) -> {
            let id = manager.add_task(title, priority)
            print("Added task #{id}: {title}")
        }
        Command.Complete(id) -> {
            match manager.complete_task(id) {
                Ok(())   -> print("Task #{id} completed! ✅")
                Err(msg) -> print("Error: {msg}")
            }
        }
        Command.Remove(id) -> {
            match manager.remove_task(id) {
                Ok(task) -> print("Removed: {task.title}")
                Err(msg) -> print("Error: {msg}")
            }
        }
        Command.List        -> manager.list_all()
        Command.ListPending -> {
            for task in manager.list_pending() {
                print(task.display())
            }
        }
        Command.Help -> {
            print("Commands:")
            print("  add <title>    — Add a task")
            print("  done <id>      — Complete a task")
            print("  remove <id>    — Remove a task")
            print("  list           — Show all tasks")
            print("  pending        — Show pending tasks")
            print("  help           — Show this help")
            print("  quit           — Exit")
        }
        Command.Quit -> {
            print("Goodbye! 👋")
            return false  // Signal to stop
        }
    }
    return true  // Continue running
}

Chapter 6: Traits and Generics

trait Searchable {
    fn matches(self, query: String) -> Bool
}

impl Searchable for Task {
    fn matches(self, query: String) -> Bool {
        let q = query.to_lower()
        return self.title.to_lower().contains(q)
            or self.description.to_lower().contains(q)
    }
}

// Generic search function — works with any Searchable type
fn search<T: Searchable>(items: &List<T>, query: String) -> List<&T> {
    return items.iter()
        .filter(fn(item) { item.matches(query) })
        .collect()
}

// Extend TaskManager with search
impl TaskManager {
    fn search(self, query: String) -> List<&Task> {
        return search(&self.tasks, query)
    }
}

Chapter 7: Error Handling

use flux.io.{File, IoError}
use flux.json

enum AppError {
    Io(IoError),
    Parse(String),
    NotFound(String),
}

impl From<IoError> for AppError {
    fn from(e: IoError) -> AppError = AppError.Io(e)
}

impl Display for AppError {
    fn to_string(self) -> String {
        match self {
            AppError.Io(e)       -> "I/O error: {e}"
            AppError.Parse(msg)  -> "Parse error: {msg}"
            AppError.NotFound(s) -> "Not found: {s}"
        }
    }
}

impl TaskManager {
    fn save(self, path: String) -> Result<Void, AppError> {
        let data = json.stringify(self.tasks)?
        File.write(path, data)?
        return Ok(())
    }

    fn load(path: String) -> Result<TaskManager, AppError> {
        let content = File.read(path)?
        let tasks: List<Task> = json.parse(content)?
        let next_id = tasks.iter()
            .map(fn(t) { t.id })
            .max()
            .unwrap_or(0) + 1

        return Ok(TaskManager { tasks: tasks, next_id: next_id })
    }
}

Chapter 9: The Complete Application

Putting it all together — a REPL loop:

use flux.io.{read_line, File}

const SAVE_FILE = "tasks.json"

fn main() {
    // Try to load existing tasks
    mut manager = match TaskManager.load(SAVE_FILE) {
        Ok(m)  -> {
            print("Loaded {m.tasks.length} task(s)")
            m
        }
        Err(_) -> TaskManager.new()
    }

    print("TaskFlux ⚡ — Type 'help' for commands\n")

    loop {
        let input = read_line("taskflux> ")

        if input.trim().is_empty() {
            continue
        }

        match parse_command(input) {
            Ok(cmd) -> {
                let should_continue = execute_command(&mut manager, cmd)

                // Auto-save after changes
                match manager.save(SAVE_FILE) {
                    Ok(()) -> {}
                    Err(e) -> print("Warning: could not save — {e}")
                }

                if not should_continue {
                    break
                }
            }
            Err(msg) -> print("Error: {msg}")
        }
    }
}
