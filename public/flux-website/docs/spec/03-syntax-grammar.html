Flux Language Specification — Syntax Grammar

Version: 0.1.0-draft
Status: Draft

1. Overview

This document defines the complete syntax of the Flux programming language using Parsing Expression Grammar (PEG) notation. Every syntactic construct is specified with its formal grammar rule, a description, and one or more code examples.

2. Program Structure

program         = module_comment* item*
item            = use_decl / fn_decl / struct_decl / class_decl / enum_decl
                / trait_decl / interface_decl / impl_block / type_alias
                / const_decl / static_decl / mod_decl / extern_block
                / macro_decl / test_decl

A Flux program is a sequence of top-level items. There is no required ordering, but the convention is:

  1. Module documentation comments (//!)
  2. Imports (use)
  3. Constants and type aliases
  4. Types (structs, classes, enums)
  5. Traits and interfaces
  6. Implementations
  7. Functions
//! A simple user management module.

use flux.collections.HashMap
use flux.io.print

const MAX_USERS = 1000

struct User {
    name: String
    email: String
}

fn main() {
    let user = User { name: "Alice", email: "alice@flux.dev" }
    print("Welcome, {user.name}!")
}

3. Declarations

3.1 Variable Declarations

let_decl    = 'let' pattern (':' type)? ('=' expression)?
mut_decl    = 'mut' IDENT (':' type)? ('=' expression)?
const_decl  = 'const' IDENT (':' type)? '=' expression
static_decl = 'static' 'mut'? IDENT ':' type '=' expression

Variables are immutable by default. Use mut for mutable bindings.

let x = 42                    // Immutable, type inferred as Int
let y: Float64 = 3.14         // Immutable, explicit type
mut counter = 0               // Mutable
const PI = 3.14159265358979   // Compile-time constant
static mut GLOBAL_COUNT: Int = 0  // Mutable static (requires unsafe to access)

Destructuring

let (x, y) = get_coordinates()          // Tuple destructuring
let (first, ...rest) = [1, 2, 3, 4]     // Array destructuring
let User { name, email } = get_user()   // Struct destructuring
let { name, email: mail } = get_user()  // With renaming

3.2 Function Declarations

fn_decl     = attribute* visibility? 'async'? 'fn' IDENT generic_params?
              '(' param_list? ')' ('->' type)? where_clause? block
param_list  = param (',' param)* ','?
param       = IDENT ':' type ('=' expression)?
visibility  = 'pub' ('(' visibility_scope ')')?
visibility_scope = 'crate' / 'super' / 'mod'

// Basic function
fn add(a: Int, b: Int) -> Int {
    return a + b
}

// Single-expression function
fn double(x: Int) -> Int = x * 2

// Default parameters
fn greet(name: String, greeting: String = "Hello") -> String {
    return "{greeting}, {name}!"
}

// Named arguments at call site
greet(name: "World", greeting: "Hi")

// Generic function
fn first<T>(items: List<T>) -> Option<T> {
    return items.get(0)
}

// Async function
async fn fetch(url: String) -> Result<String, HttpError> {
    let response = await http.get(url)?
    return Ok(response.body)
}

// Public function
pub fn public_api() -> Int {
    return 42
}

// Function with trait bounds
fn print_all<T>(items: List<T>) where T: Printable {
    for item in items {
        print(item.to_string())
    }
}

Return Type

  • If the return type is omitted, it defaults to Void (no return value).
  • The return keyword is required for explicit returns. The last expression in a block is not implicitly returned (unlike Rust). Use return for clarity.
  • Exception: single-expression functions using = expression syntax return the expression's value.
// Explicit return required
fn compute(x: Int) -> Int {
    let result = x * 2 + 1
    return result
}

// Single-expression shorthand — no `return` needed
fn square(x: Int) -> Int = x * x

// Void function — no return type
fn log(message: String) {
    print("[LOG] {message}")
}

3.3 Struct Declarations

struct_decl = attribute* visibility? 'struct' IDENT generic_params? where_clause?
              '{' struct_field* '}'
struct_field = attribute* visibility? IDENT ':' type ('=' expression)?

Structs are value types with no inheritance. They are allocated on the stack by default.

struct Point {
    x: Float64
    y: Float64
}

struct Color {
    r: UInt8
    g: UInt8
    b: UInt8
    a: UInt8 = 255  // Default value
}

// Generic struct
struct Pair<A, B> {
    first: A
    second: B
}

// Instantiation
let origin = Point { x: 0.0, y: 0.0 }
let red = Color { r: 255, g: 0, b: 0 }  // a defaults to 255
let pair = Pair { first: "hello", second: 42 }

3.4 Class Declarations

class_decl = attribute* visibility? 'class' IDENT generic_params?
             (':' type_list)? where_clause? '{' class_member* '}'
class_member = struct_field / fn_decl / init_decl / deinit_decl
init_decl    = visibility? 'init' '(' param_list? ')' block
deinit_decl  = 'deinit' block

Classes are reference types with single inheritance. They are heap-allocated and managed by ARC (Automatic Reference Counting).

class Animal {
    name: String
    sound: String

    init(name: String, sound: String) {
        self.name = name
        self.sound = sound
    }

    fn speak(self) -> String {
        return "{self.name} says {self.sound}!"
    }

    deinit {
        print("{self.name} is being deallocated")
    }
}

// Inheritance
class Dog: Animal {
    breed: String

    init(name: String, breed: String) {
        super.init(name: name, sound: "Woof")
        self.breed = breed
    }

    fn fetch(self, item: String) -> String {
        return "{self.name} fetches the {item}!"
    }
}

let dog = Dog(name: "Rex", breed: "Labrador")
print(dog.speak())     // "Rex says Woof!"
print(dog.fetch("ball")) // "Rex fetches the ball!"

3.5 Enum Declarations

enum_decl    = attribute* visibility? 'enum' IDENT generic_params? where_clause?
               '{' enum_variant (',' enum_variant)* ','? '}'
enum_variant = IDENT enum_variant_data?
enum_variant_data = '(' field_list ')' / '{' struct_field* '}'
field_list   = field (',' field)* ','?
field        = (IDENT ':')? type

Enums are algebraic data types (sum types). Each variant can optionally carry data.

// Simple enum
enum Direction {
    North,
    South,
    East,
    West,
}

// Enum with associated data
enum Shape {
    Circle(radius: Float64),
    Rectangle(width: Float64, height: Float64),
    Triangle(base: Float64, height: Float64),
    Point,
}

// Generic enum
enum Option<T> {
    Some(T),
    None,
}

enum Result<T, E> {
    Ok(T),
    Err(E),
}

// Complex enum
enum Token {
    Integer(value: Int),
    Float(value: Float64),
    String(value: String),
    Identifier(name: String),
    Keyword(kind: KeywordKind),
    Eof,
}

// Usage
let shape = Shape.Circle(radius: 5.0)
let result: Result<Int, String> = Ok(42)

3.6 Trait Declarations

trait_decl    = attribute* visibility? 'trait' IDENT generic_params?
               (':' type_list)? where_clause? '{' trait_member* '}'
trait_member  = fn_signature / fn_decl / type_alias / const_decl
fn_signature  = 'fn' IDENT generic_params? '(' param_list? ')' ('->' type)? where_clause?

Traits define shared behavior. They can contain: - Required methods (signature only) - Default methods (with implementation) - Associated types - Associated constants

trait Printable {
    fn to_string(self) -> String
}

trait Comparable: Equatable {
    fn compare(self, other: Self) -> Ordering

    // Default methods
    fn less_than(self, other: Self) -> Bool {
        return self.compare(other) == Ordering.Less
    }

    fn greater_than(self, other: Self) -> Bool {
        return self.compare(other) == Ordering.Greater
    }
}

trait Iterator {
    type Item  // Associated type

    fn next(mut self) -> Option<Self.Item>

    // Default method using the required method
    fn count(mut self) -> Int {
        mut n = 0
        while self.next() is Some(_) {
            n += 1
        }
        return n
    }
}

trait Addable {
    const ZERO: Self  // Associated constant

    fn add(self, other: Self) -> Self
}

3.7 Interface Declarations

interface_decl = attribute* visibility? 'interface' IDENT generic_params?
                 '{' fn_signature* '}'

Interfaces are like traits but cannot have default implementations or associated types. They define a pure contract. Interfaces support structural subtyping — a type satisfies an interface if it has all the required methods, even without an explicit impl declaration.

interface Serializable {
    fn serialize(self) -> Bytes
    fn deserialize(data: Bytes) -> Self
}

interface Cloneable {
    fn clone(self) -> Self
}

3.8 Implementation Blocks

impl_block = 'impl' generic_params? type ('for' type)? where_clause? '{' fn_decl* '}'

// Methods on a type
impl Point {
    fn distance_to(self, other: Point) -> Float64 {
        let dx = self.x - other.x
        let dy = self.y - other.y
        return (dx * dx + dy * dy).sqrt()
    }

    fn translate(self, dx: Float64, dy: Float64) -> Point {
        return Point { x: self.x + dx, y: self.y + dy }
    }
}

// Implementing a trait for a type
impl Printable for Point {
    fn to_string(self) -> String {
        return "({self.x}, {self.y})"
    }
}

// Generic implementation
impl<T: Printable> Printable for List<T> {
    fn to_string(self) -> String {
        let items = self.map(fn(item) { item.to_string() }).join(", ")
        return "[{items}]"
    }
}

3.9 Type Aliases

type_alias = 'type' IDENT generic_params? '=' type

type StringMap<V> = HashMap<String, V>
type Callback = fn(Int, Int) -> Int
type Result<T> = flux.Result<T, AppError>

3.10 Module Declarations

mod_decl = visibility? 'mod' IDENT ('{' item* '}' / ';')

Modules organize code into namespaces. A module can be declared inline or in a separate file.

// Inline module
mod math {
    pub fn abs(x: Int) -> Int {
        if x < 0 { return -x }
        return x
    }
}

// File module — compiler looks for `utils.flux` or `utils/mod.flux`
mod utils

3.11 Use Declarations (Imports)

use_decl    = 'use' use_path
use_path    = path_segment ('.' path_segment)* use_suffix?
use_suffix  = '.{' use_item (',' use_item)* ','? '}'
            / '.' '*'
            / 'as' IDENT
use_item    = IDENT ('as' IDENT)?

use flux.io.print                     // Import a single item
use flux.collections.{HashMap, List}  // Import multiple items
use flux.math.*                       // Import everything (glob)
use flux.http.Client as HttpClient    // Import with alias
use flux.json.{parse, stringify as to_json}  // Mixed

4. Expressions

4.1 Literal Expressions

literal_expr = integer_literal / float_literal / string_literal
             / char_literal / bool_literal / none_literal
             / array_literal / map_literal / tuple_literal

See Lexical Structure §7 for literal formats.

4.2 Identifier Expressions

ident_expr = IDENT ('.' IDENT)*

x
user.name
flux.io.print

4.3 Unary Expressions

unary_expr = ('-' / 'not' / '~' / '&' / '&mut' / '*') expression

-x          // Negation
not enabled // Logical NOT
~bits       // Bitwise NOT
&value      // Shared reference
&mut value  // Mutable reference
*ptr        // Dereference

4.4 Binary Expressions

binary_expr = expression operator expression

All binary operators and their precedence are defined in Lexical Structure §8.

4.5 Function Call Expressions

call_expr = expression '(' arg_list? ')'
arg_list  = arg (',' arg)* ','?
arg       = (IDENT ':')? expression

print("Hello")
add(1, 2)
greet(name: "Alice", greeting: "Hi")
user.method()
list.map(fn(x) { x * 2 })

4.6 Index Expressions

index_expr = expression '[' expression ']'

let first = array[0]
let value = map["key"]
matrix[row][col] = 42

4.7 Member Access Expressions

member_expr = expression '.' IDENT
optional_chain_expr = expression '?.' IDENT

user.name
point.x
user?.address?.city    // Returns none if any step is none

4.8 If Expressions

if_expr = 'if' expression block ('else' 'if' expression block)* ('else' block)?

if is an expression — it returns a value:

// As a statement
if temperature > 100 {
    print("Too hot!")
} else if temperature < 0 {
    print("Too cold!")
} else {
    print("Just right.")
}

// As an expression
let status = if score >= 90 {
    "A"
} else if score >= 80 {
    "B"
} else {
    "C"
}

// If-let for Option unwrapping
if let Some(user) = find_user(id) {
    print("Found: {user.name}")
} else {
    print("User not found")
}

4.9 Match Expressions

match_expr    = 'match' expression '{' match_arm+ '}'
match_arm     = pattern ('if' expression)? '->' (expression / block)
pattern       = literal_pattern / ident_pattern / tuple_pattern
              / struct_pattern / enum_pattern / wildcard_pattern
              / range_pattern / or_pattern / binding_pattern
wildcard_pattern = '_'
or_pattern       = pattern ('|' pattern)+
binding_pattern  = IDENT '@' pattern

Pattern matching must be exhaustive — all possible values must be covered.

// Simple value matching
match direction {
    Direction.North -> print("Going north")
    Direction.South -> print("Going south")
    Direction.East  -> print("Going east")
    Direction.West  -> print("Going west")
}

// Matching with data extraction
match shape {
    Shape.Circle(r)        -> 3.14159 * r * r
    Shape.Rectangle(w, h)  -> w * h
    Shape.Triangle(b, h)   -> 0.5 * b * h
    Shape.Point            -> 0.0
}

// Guard clauses
match value {
    x if x < 0    -> print("Negative: {x}")
    x if x == 0   -> print("Zero")
    x if x < 100  -> print("Small positive: {x}")
    x              -> print("Large: {x}")
}

// Nested patterns
match result {
    Ok(Some(value)) -> print("Got: {value}")
    Ok(None)        -> print("Empty result")
    Err(error)      -> print("Error: {error}")
}

// Or patterns
match key {
    'a' | 'e' | 'i' | 'o' | 'u' -> print("Vowel")
    _                              -> print("Consonant")
}

// Range patterns
match age {
    0..=12   -> "Child"
    13..=17  -> "Teenager"
    18..=64  -> "Adult"
    _        -> "Senior"
}

// Binding patterns
match value {
    n @ 1..=100 -> print("In range: {n}")
    n           -> print("Out of range: {n}")
}

// Struct patterns
match point {
    Point { x: 0, y: 0 }     -> print("Origin")
    Point { x, y: 0 }        -> print("On x-axis at {x}")
    Point { x: 0, y }        -> print("On y-axis at {y}")
    Point { x, y }            -> print("At ({x}, {y})")
}

4.10 Closure Expressions

closure_expr = ('fn' / 'move')? '(' param_list? ')' ('->' type)? block
             / ('fn' / 'move')? '(' param_list? ')' '=>' expression

// Full closure
let add = fn(a: Int, b: Int) -> Int {
    return a + b
}

// Arrow closure (single expression)
let double = (x: Int) => x * 2

// Type-inferred parameters
let square = (x) => x * x

// Used inline
let evens = numbers.filter((n) => n % 2 == 0)

// Multi-line closure
let process = fn(items: List<Int>) -> List<Int> {
    mut result = List<Int>.new()
    for item in items {
        if item > 0 {
            result.append(item * 2)
        }
    }
    return result
}

// Move closure (takes ownership of captured variables)
let name = "Alice"
let greet = move fn() -> String {
    return "Hello, {name}!"
}

4.11 Block Expressions

block = '{' statement* '}'

A block is a sequence of statements enclosed in curly braces. Blocks create a new scope.

let result = {
    let a = compute_a()
    let b = compute_b()
    a + b   // Note: blocks don't implicitly return; use `return` in functions
}

4.12 Tuple Expressions

tuple_expr = '(' expression ',' expression (',' expression)* ','? ')'

let point = (10, 20)
let (x, y) = point
print(point.0)   // Access by index

let named = (x: 10, y: 20)
print(named.x)   // Access by name

4.13 Array Expressions

array_expr = '[' (expression (',' expression)* ','?)? ']'
           / '[' expression ';' expression ']'

let numbers = [1, 2, 3, 4, 5]
let zeros = [0; 100]              // 100 zeros
let matrix = [[1, 2], [3, 4]]    // Nested arrays

4.14 Map Expressions

map_expr = '{' (map_entry (',' map_entry)* ','?)? '}'
map_entry = expression ':' expression

let scores = {
    "Alice": 95
    "Bob": 87
    "Carol": 92
}

4.15 Range Expressions

range_expr = expression '..' expression
           / expression '..=' expression

for i in 0..10 {       // 0 to 9
    print(i)
}

for i in 0..=10 {      // 0 to 10
    print(i)
}

4.16 Pipe Expressions

pipe_expr = expression '|>' expression

The pipe operator passes the left-hand side as the first argument to the right-hand side function.

let result = data
    |> parse
    |> validate
    |> transform
    |> serialize

// Equivalent to:
let result = serialize(transform(validate(parse(data))))

4.17 Type Cast Expressions

cast_expr  = expression 'as' type
check_expr = expression 'is' type

let x: Float64 = 42 as Float64
let n: Int = 3.14 as Int         // Truncates to 3

if value is String {
    print("It's a string: {value as String}")
}

4.18 None Coalescing and Optional Chaining

coalesce_expr = expression '??' expression
chain_expr    = expression '?.' IDENT

let name = user.name ?? "Anonymous"
let city = user?.address?.city ?? "Unknown"

4.19 Await Expressions

await_expr = 'await' expression

let response = await http.get("https://api.example.com")
let (a, b) = await (fetch_a(), fetch_b())  // Concurrent await

5. Statements

5.1 Expression Statements

Any expression can be used as a statement.

print("hello")
x + y           // Warning: result unused
user.save()

5.2 Assignment Statements

assignment = expression '=' expression
           / expression compound_assign expression
compound_assign = '+=' / '-=' / '*=' / '/=' / '%=' / '**='
                / '&=' / '|=' / '^=' / '<<=' / '>>='

mut x = 10
x = 20
x += 5
x *= 2

5.3 Return Statements

return_stmt = 'return' expression?

fn compute() -> Int {
    return 42
}

fn do_nothing() {
    return   // Returns Void
}

5.4 Defer Statements

defer_stmt = 'defer' (block / expression)

defer schedules code to run when the enclosing scope exits, regardless of how it exits (return, error, etc.). Multiple defers execute in LIFO (reverse) order.

fn read_file(path: String) -> Result<String, IoError> {
    let file = File.open(path)?
    defer file.close()

    return file.read_all()
}

fn transaction() {
    db.begin()
    defer db.rollback()  // Runs if we don't commit

    // ... operations ...

    db.commit()
}

6. Control Flow

6.1 For Loops

for_stmt = 'for' pattern 'in' expression block

// Iterate over a range
for i in 0..10 {
    print(i)
}

// Iterate over a collection
for item in items {
    print(item)
}

// Destructuring in for loop
for (index, value) in items.enumerate() {
    print("{index}: {value}")
}

// Iterate over map entries
for (key, value) in scores {
    print("{key} = {value}")
}

// For with filter (syntactic sugar)
for item in items if item.is_valid() {
    process(item)
}

6.2 While Loops

while_stmt = 'while' expression block
while_let  = 'while' 'let' pattern '=' expression block

mut count = 0
while count < 10 {
    print(count)
    count += 1
}

// While-let
while let Some(item) = iterator.next() {
    process(item)
}

6.3 Loop (Infinite)

loop_stmt = 'loop' block

loop {
    let input = read_line()
    if input == "quit" {
        break
    }
    process(input)
}

// Loop with value (break returns a value)
let result = loop {
    let attempt = try_operation()
    if attempt.is_ok() {
        break attempt.unwrap()
    }
}

6.4 Break and Continue

break_stmt    = 'break' expression?
continue_stmt = 'continue'

for item in items {
    if item.is_invalid() {
        continue     // Skip this iteration
    }
    if item.is_final() {
        break        // Exit the loop
    }
    process(item)
}

// Labeled loops
outer: for i in 0..10 {
    for j in 0..10 {
        if i * j > 50 {
            break outer  // Break the outer loop
        }
    }
}

7. Generics

generic_params = '<' generic_param (',' generic_param)* ','? '>'
generic_param  = IDENT (':' type_bound)?
type_bound     = type ('+' type)*
where_clause   = 'where' where_predicate (',' where_predicate)* ','?
where_predicate = type ':' type_bound

// Generic function
fn max<T: Comparable>(a: T, b: T) -> T {
    if a.greater_than(b) {
        return a
    }
    return b
}

// Generic struct
struct Stack<T> {
    items: List<T>
}

impl<T> Stack<T> {
    fn new() -> Stack<T> {
        return Stack { items: List.new() }
    }

    fn push(mut self, item: T) {
        self.items.append(item)
    }

    fn pop(mut self) -> Option<T> {
        return self.items.pop()
    }
}

// Multiple bounds
fn process<T: Printable + Comparable>(item: T) {
    print(item.to_string())
}

// Where clause (for complex bounds)
fn merge<A, B, C>(a: A, b: B) -> C
    where A: Into<C>,
          B: Into<C>,
          C: Addable {
    return a.into().add(b.into())
}

// Generic enum
enum Either<L, R> {
    Left(L),
    Right(R),
}

8. Type Annotations

type        = simple_type / generic_type / function_type / tuple_type
            / array_type / reference_type / optional_type
simple_type = IDENT ('.' IDENT)*
generic_type = simple_type '<' type (',' type)* '>'
function_type = 'fn' '(' type_list? ')' '->' type
tuple_type   = '(' type ',' type (',' type)* ')'
array_type   = '[' type ']'
reference_type = '&' 'mut'? type
optional_type  = type '?'

let x: Int = 42
let name: String = "Flux"
let items: List<Int> = [1, 2, 3]
let callback: fn(Int, Int) -> Int = add
let point: (Float64, Float64) = (1.0, 2.0)
let numbers: [Int] = [1, 2, 3]
let reference: &String = &name
let optional: Int? = none     // Shorthand for Option<Int>
let map: HashMap<String, List<Int>> = HashMap.new()

9. Attributes and Annotations

attribute    = '#[' attr_path attr_args? ']'
attr_path    = IDENT ('.' IDENT)*
attr_args    = '(' attr_arg_list ')'
attr_arg_list = attr_arg (',' attr_arg)*
attr_arg     = IDENT '=' literal / IDENT / literal

#[derive(Debug, Clone, Eq, Hash)]
struct Point {
    x: Float64
    y: Float64
}

#[deprecated(since: "2.0", note: "Use new_function instead")]
fn old_function() {
    // ...
}

#[cfg(target_os: "linux")]
fn linux_specific() {
    // Only compiled on Linux
}

#[inline]
fn fast_function(x: Int) -> Int {
    return x * 2
}

#[test]
fn test_addition() {
    assert(1 + 1 == 2)
}

#[bench]
fn bench_sort(b: Bencher) {
    b.iter(fn() {
        let data = generate_random_data(1000)
        data.sort()
    })
}

10. Test Declarations

test_decl = '#[test]' fn_decl
          / 'test' STRING_LITERAL block

Flux supports two styles of test declarations:

// Attribute style (preferred for unit tests)
#[test]
fn test_addition() {
    assert(add(2, 3) == 5)
    assert(add(-1, 1) == 0)
}

// Block style (for descriptive integration tests)
test "user creation should validate email" {
    let result = User.create(name: "Alice", email: "invalid")
    assert(result.is_err())
    assert(result.unwrap_err().message.contains("email"))
}

// Test with expected failure
#[test]
#[should_panic(message: "division by zero")]
fn test_division_by_zero() {
    let _ = 1 / 0
}

11. Extern Blocks

extern_block = 'extern' STRING_LITERAL '{' extern_item* '}'
extern_item  = 'fn' IDENT '(' param_list? ')' ('->' type)?

extern "C" {
    fn printf(format: *Byte, ...) -> Int
    fn malloc(size: USize) -> *mut Byte
    fn free(ptr: *mut Byte)
}

12. Operator Overloading

Operators are overloaded by implementing the corresponding trait:

Operator Trait Method
+ Add fn add(self, rhs: Rhs) -> Output
- Sub fn sub(self, rhs: Rhs) -> Output
* Mul fn mul(self, rhs: Rhs) -> Output
/ Div fn div(self, rhs: Rhs) -> Output
% Rem fn rem(self, rhs: Rhs) -> Output
== Eq fn eq(self, rhs: Rhs) -> Bool
< Ord fn cmp(self, rhs: Rhs) -> Ordering
[] Index fn index(self, idx: Idx) -> Output
- (unary) Neg fn neg(self) -> Output
~ BitNot fn bit_not(self) -> Output 
impl Add for Point {
    type Output = Point

    fn add(self, rhs: Point) -> Point {
        return Point { x: self.x + rhs.x, y: self.y + rhs.y }
    }
}

let a = Point { x: 1.0, y: 2.0 }
let b = Point { x: 3.0, y: 4.0 }
let c = a + b  // Point { x: 4.0, y: 6.0 }

13. Extension Methods

extend_block = 'extend' type where_clause? '{' fn_decl* '}'

Extension methods add new methods to existing types without modifying their source:

extend String {
    fn is_blank(self) -> Bool {
        return self.trim().is_empty()
    }

    fn repeat(self, n: Int) -> String {
        mut result = ""
        for _ in 0..n {
            result += self
        }
        return result
    }
}

let s = "hello"
print(s.repeat(3))    // "hellohellohello"
print("  ".is_blank()) // true

14. Async and Concurrency Syntax

14.1 Async Functions

async fn fetch_user(id: Int) -> Result<User, ApiError> {
    let response = await http.get("/api/users/{id}")?
    let user: User = json.parse(response.body)?
    return Ok(user)
}

14.2 Spawn Tasks

async fn main() {
    let task1 = spawn fetch_user(1)
    let task2 = spawn fetch_user(2)

    let user1 = await task1
    let user2 = await task2

    print("{user1.name}, {user2.name}")
}

14.3 Task Groups

async fn fetch_all_users(ids: List<Int>) -> List<User> {
    let group = TaskGroup<User>.new()

    for id in ids {
        group.spawn(async {
            return await fetch_user(id)
        })
    }

    return await group.collect()
}

14.4 Channels

async fn producer_consumer() {
    let (sender, receiver) = Channel<Int>.new(buffer: 10)

    spawn async {
        for i in 0..100 {
            await sender.send(i)
        }
        sender.close()
    }

    while let Some(value) = await receiver.receive() {
        print("Received: {value}")
    }
}

15. Error Handling Syntax

15.1 Result and Option

fn divide(a: Int, b: Int) -> Result<Int, MathError> {
    if b == 0 {
        return Err(MathError.DivisionByZero)
    }
    return Ok(a / b)
}

fn find_user(id: Int) -> Option<User> {
    return database.users.get(id)
}

15.2 Error Propagation (?)

fn process_file(path: String) -> Result<Data, AppError> {
    let content = File.read(path)?        // Propagates IoError
    let parsed = json.parse(content)?     // Propagates ParseError
    let validated = validate(parsed)?     // Propagates ValidationError
    return Ok(validated)
}

15.3 Try-Catch

try/catch is syntactic sugar over match on Result:

try {
    let file = File.open("data.txt")?
    let content = file.read_all()?
    process(content)
} catch IoError.NotFound {
    print("File not found")
} catch IoError.PermissionDenied {
    print("Permission denied")
} catch error {
    print("Unexpected error: {error}")
}

16. Complete Formal Grammar Summary

// === Program ===
program         = item*
item            = use_decl / fn_decl / struct_decl / class_decl / enum_decl
                / trait_decl / interface_decl / impl_block / type_alias
                / const_decl / static_decl / mod_decl / extern_block / test_decl

// === Declarations ===
let_decl        = 'let' pattern (':' type)? '=' expression
mut_decl        = 'mut' IDENT (':' type)? '=' expression
const_decl      = 'const' IDENT (':' type)? '=' expression
fn_decl         = attribute* visibility? 'async'? 'fn' IDENT generic_params?
                  '(' param_list? ')' ('->' type)? where_clause? (block / '=' expression)
struct_decl     = attribute* visibility? 'struct' IDENT generic_params? '{' struct_field* '}'
class_decl      = attribute* visibility? 'class' IDENT generic_params? (':' type_list)?
                  '{' class_member* '}'
enum_decl       = attribute* visibility? 'enum' IDENT generic_params? '{' enum_variant+ '}'
trait_decl      = attribute* visibility? 'trait' IDENT generic_params? (':' type_list)?
                  '{' trait_member* '}'
interface_decl  = attribute* visibility? 'interface' IDENT generic_params? '{' fn_signature* '}'
impl_block      = 'impl' generic_params? type ('for' type)? where_clause? '{' fn_decl* '}'
type_alias      = 'type' IDENT generic_params? '=' type
mod_decl        = visibility? 'mod' IDENT ('{' item* '}' / ';')
use_decl        = 'use' use_path
extern_block    = 'extern' STRING '{' extern_item* '}'
extend_block    = 'extend' type '{' fn_decl* '}'

// === Expressions ===
expression      = assignment_expr
assignment_expr = pipe_expr (('=' / '+=' / '-=' / '*=' / '/=' / '%=') pipe_expr)?
pipe_expr       = or_expr ('|>' or_expr)*
or_expr         = and_expr ('or' and_expr)*
and_expr        = equality_expr ('and' equality_expr)*
equality_expr   = comparison_expr (('==' / '!=') comparison_expr)?
comparison_expr = range_expr (('<' / '>' / '<=' / '>=') range_expr)?
range_expr      = additive_expr (('..' / '..=') additive_expr)?
additive_expr   = multiplicative_expr (('+' / '-') multiplicative_expr)*
multiplicative_expr = power_expr (('*' / '/' / '%') power_expr)*
power_expr      = unary_expr ('**' unary_expr)?
unary_expr      = ('-' / 'not' / '~' / '&' / '&mut' / '*') unary_expr / postfix_expr
postfix_expr    = primary_expr (call_suffix / index_suffix / member_suffix / '?' / 'as' type / 'is' type)*
primary_expr    = literal / IDENT / '(' expression ')' / if_expr / match_expr
                / block / closure_expr / array_expr / map_expr / 'await' expression

// === Statements ===
statement       = let_decl / mut_decl / expression / return_stmt / defer_stmt
                / for_stmt / while_stmt / loop_stmt / break_stmt / continue_stmt

// === Control Flow ===
if_expr         = 'if' ('let' pattern '=')? expression block ('else' (if_expr / block))?
match_expr      = 'match' expression '{' match_arm+ '}'
match_arm       = pattern ('if' expression)? '->' (expression / block)
for_stmt        = 'for' pattern 'in' expression ('if' expression)? block
while_stmt      = 'while' ('let' pattern '=')? expression block
loop_stmt       = 'loop' block

// === Types ===
type            = IDENT ('<' type_list '>')? / fn_type / tuple_type / array_type
                / '&' 'mut'? type / type '?'
fn_type         = 'fn' '(' type_list? ')' '->' type
tuple_type      = '(' type ',' type_list ')'
array_type      = '[' type ']'

// === Patterns ===
pattern         = '_' / literal / IDENT ('@' pattern)? / '(' pattern_list ')'
                / IDENT '{' field_pattern_list '}' / IDENT '.' IDENT ('(' pattern_list ')')?
                / pattern '|' pattern / pattern '..' pattern

This grammar serves as the definitive reference for the Flux parser implementation. All ambiguities are resolved by operator precedence (§8.8 of the Lexical Structure) and the ordered-choice semantics of PEG.
