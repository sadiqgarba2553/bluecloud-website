Flux Language Specification — Interoperability

Version: 0.1.0-draft
Status: Draft

1. Overview

Flux is designed to interoperate seamlessly with existing ecosystems. It provides first-class support for calling C code, and varying levels of support for C++, Python, JavaScript (via WebAssembly), and platform-native APIs.

2. C FFI

2.1 Calling C Functions

extern "C" {
    fn printf(format: *Byte, ...) -> Int
    fn strlen(s: *Byte) -> USize
    fn malloc(size: USize) -> *mut Byte
    fn free(ptr: *mut Byte)
    fn qsort(
        base: *mut Byte
        nmemb: USize
        size: USize
        compar: fn(*Byte, *Byte) -> Int
    )
}

fn main() {
    unsafe {
        let s = "Hello from Flux!\0"
        printf(s.as_ptr(), s.len())
    }
}

2.2 Linking

// Link to a system library
#[link(name: "sqlite3")]
extern "C" {
    fn sqlite3_open(filename: *Byte, db: *mut *mut SqliteDb) -> Int
    fn sqlite3_close(db: *mut SqliteDb) -> Int
    fn sqlite3_exec(
        db: *mut SqliteDb
        sql: *Byte
        callback: Option<fn(*Byte, Int, *mut *Byte, *mut *Byte) -> Int>
        arg: *mut Byte
        errmsg: *mut *Byte
    ) -> Int
}

// Link to a static library
#[link(name: "mylib", kind: "static")]
extern "C" {
    fn my_function(x: Int) -> Int
}

// Link to a framework (macOS/iOS)
#[link(name: "CoreFoundation", kind: "framework")]
extern "C" {
    fn CFRelease(cf: *Byte)
}

2.4 Exposing Flux Functions to C

#[no_mangle]
pub extern "C" fn flux_add(a: Int32, b: Int32) -> Int32 {
    return a + b
}

#[no_mangle]
pub extern "C" fn flux_process(data: *Byte, len: USize) -> Int32 {
    unsafe {
        let slice = Slice.from_raw(data, len)
        // Process the data...
        return 0
    }
}

2.5 Struct Layout

// C-compatible struct layout
#[repr(C)]
struct CPoint {
    x: Float64
    y: Float64
}

// Packed struct (no padding)
#[repr(C, packed)]
struct PackedHeader {
    magic: UInt32
    version: UInt16
    flags: UInt16
}

// Enum with specific representation
#[repr(Int32)]
enum CError {
    Success = 0,
    NotFound = -1,
    PermissionDenied = -2,
    IoError = -3,
}

2.6 Callbacks

extern "C" {
    fn register_callback(cb: extern "C" fn(Int32) -> Int32)
}

extern "C" fn my_callback(value: Int32) -> Int32 {
    return value * 2
}

fn main() {
    unsafe {
        register_callback(my_callback)
    }
}

2.7 Safe Wrappers

Best practice: wrap unsafe FFI calls in safe Flux APIs:

// Low-level FFI (private)
#[link(name: "sqlite3")]
extern "C" {
    fn sqlite3_open(filename: *Byte, db: *mut *mut RawDb) -> Int32
    fn sqlite3_close(db: *mut RawDb) -> Int32
    fn sqlite3_errmsg(db: *mut RawDb) -> *Byte
}

// Safe public API
pub struct Database {
    handle: *mut RawDb
}

impl Database {
    pub fn open(path: String) -> Result<Database, DbError> {
        let mut handle: *mut RawDb = null_mut()
        let cpath = CString.from(path)

        let result = unsafe { sqlite3_open(cpath.as_ptr(), &mut handle) }

        if result != 0 {
            let msg = unsafe { CString.from_ptr(sqlite3_errmsg(handle)) }
            return Err(DbError.OpenFailed(msg.to_string()))
        }

        return Ok(Database { handle: handle })
    }

    impl Drop {
        fn drop(mut self) {
            unsafe { sqlite3_close(self.handle) }
        }
    }
}

3. C++ Interoperability

3.1 Calling C++ via C Wrappers

The recommended approach for C++ interop is to create C wrapper functions:

// wrapper.h
extern "C" {
    void* create_engine();
    void engine_process(void* engine, const float* data, int len);
    void destroy_engine(void* engine);
}

extern "C" {
    fn create_engine() -> *mut Byte
    fn engine_process(engine: *mut Byte, data: *Float32, len: Int32)
    fn destroy_engine(engine: *mut Byte)
}

pub struct Engine {
    handle: *mut Byte
}

impl Engine {
    pub fn new() -> Engine {
        let handle = unsafe { create_engine() }
        return Engine { handle: handle }
    }

    pub fn process(self, data: &[Float32]) {
        unsafe {
            engine_process(self.handle, data.as_ptr(), data.len() as Int32)
        }
    }

    impl Drop {
        fn drop(mut self) {
            unsafe { destroy_engine(self.handle) }
        }
    }
}

3.2 Automatic Binding Generation

The flux bindgen tool generates Flux bindings from C/C++ header files:

flux bindgen vendor/sqlite3.h --output src/bindings.flux
flux bindgen vendor/engine.hpp --output src/engine_bindings.flux --language c++

4. Python Interoperability

4.1 Calling Python from Flux

use flux.interop.python.{Python, PyObject}

fn main() {
    let py = Python.init()

    // Execute Python code
    py.run("print('Hello from Python!')")

    // Import a Python module
    let np = py.import("numpy")
    let array = np.call("array", [[1, 2, 3], [4, 5, 6]])
    let result = array.call("sum")
    print("Sum: {result.to_int()}")

    // Call a Python function
    let math = py.import("math")
    let pi = math.attr("pi").to_float()
    print("Pi: {pi}")
}

4.2 Exposing Flux to Python

// Create a Python module from Flux code
#[pymodule]
mod flux_math {
    #[pyfunction]
    fn fast_fibonacci(n: Int) -> Int {
        if n <= 1 { return n }
        mut a = 0
        mut b = 1
        for _ in 2..=n {
            let temp = a + b
            a = b
            b = temp
        }
        return b
    }

    #[pyclass]
    struct Matrix {
        data: List<List<Float64>>
        rows: Int
        cols: Int
    }

    #[pymethods]
    impl Matrix {
        #[new]
        fn new(rows: Int, cols: Int) -> Matrix {
            // ...
        }

        fn multiply(self, other: &Matrix) -> Matrix {
            // ...
        }
    }
}

Usage from Python:

import flux_math

result = flux_math.fast_fibonacci(40)  # Runs at native speed
print(f"Fibonacci(40) = {result}")

m = flux_math.Matrix(3, 3)

5. WebAssembly

5.1 Compiling to WebAssembly

flux build --target wasm32

This produces a .wasm file that can be loaded in browsers or Node.js.

5.2 WASM Exports

#[wasm_export]
pub fn add(a: Int32, b: Int32) -> Int32 {
    return a + b
}

#[wasm_export]
pub fn process_data(ptr: Int32, len: Int32) -> Int32 {
    // Process data from WebAssembly linear memory
    let data = unsafe { wasm.memory_slice(ptr as USize, len as USize) }
    // ...
    return 0
}

5.3 Calling JavaScript from WASM

#[wasm_import(module: "env")]
extern {
    fn console_log(ptr: Int32, len: Int32)
    fn fetch_url(url_ptr: Int32, url_len: Int32) -> Int32
}

fn log(message: String) {
    unsafe {
        console_log(message.as_ptr() as Int32, message.len() as Int32)
    }
}

6. Platform-Native APIs

6.1 Android (JNI)

use flux.interop.jni.{JniEnv, JObject, JString}

#[jni_export(class: "com.example.MyApp")]
fn nativeCompute(env: &JniEnv, _this: JObject, input: JString) -> JString {
    let rust_string = env.get_string(input)
    let result = process(rust_string)
    return env.new_string(result)
}

6.2 iOS/macOS (Objective-C Bridge)

use flux.interop.objc.{ObjcObject, selector, msg_send}

fn show_alert(title: String, message: String) {
    unsafe {
        let alert_class = ObjcObject.class("NSAlert")
        let alert = msg_send!(alert_class, "new")
        msg_send!(alert, "setMessageText:", title)
        msg_send!(alert, "setInformativeText:", message)
        msg_send!(alert, "runModal")
    }
}

6.3 Windows (Win32)

#[link(name: "user32")]
extern "C" {
    fn MessageBoxA(
        hwnd: *Byte
        text: *Byte
        caption: *Byte
        utype: UInt32
    ) -> Int32
}

fn show_message(title: String, message: String) {
    unsafe {
        MessageBoxA(
            null()
            CString.from(message).as_ptr()
            CString.from(title).as_ptr()
            0
        )
    }
}

7. ABI Stability

7.1 Flux ABI

Flux does not guarantee a stable ABI across compiler versions (similar to Rust). Recompilation is required when updating the compiler.

7.2 C ABI

Functions marked extern "C" use the platform's C ABI, which is stable.

7.3 Stable Interface Files

For creating stable plugin interfaces, use the #[stable_abi] attribute:

#[stable_abi]
pub trait Plugin {
    fn name(self) -> &String
    fn version(self) -> (Int, Int, Int)
    fn execute(self, input: &[Byte]) -> Result<Vec<Byte>, PluginError>
}

8. Inline Assembly

For the lowest level of control, Flux supports inline assembly:

fn read_cycle_counter() -> UInt64 {
    let result: UInt64
    unsafe {
        asm! {
            "rdtsc"
            "shl rdx, 32"
            "or rax, rdx"
            out("rax") result
            clobber("rdx")
        }
    }
    return result
}

fn memory_fence() {
    unsafe {
        asm! { "mfence" }
    }
}

8.1 Inline Assembly Syntax

asm! {
    "assembly instruction"
    in("register") value           // Input operand
    out("register") variable       // Output operand
    inout("register") variable     // Input and output
    clobber("register")            // Clobbered register
    options(nostack, nomem)        // Options
}

9. Bindgen and Tooling

9.1 flux bindgen

Automatically generates Flux bindings from C/C++ headers:

# Generate from C header
flux bindgen sqlite3.h -o src/sqlite3.flux

# With configuration
flux bindgen openssl/ssl.h \
    --allowlist-function "SSL_.*" \
    --allowlist-type "SSL_CTX" \
    --link "ssl" \
    --link "crypto" \
    -o src/openssl.flux

9.2 flux cbindgen

Generates C/C++ headers from Flux source:

# Generate C header from Flux library
flux cbindgen src/ --output include/mylib.h --language c
flux cbindgen src/ --output include/mylib.hpp --language c++
